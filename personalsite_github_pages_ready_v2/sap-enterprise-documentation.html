Technical Implementation Code

// SAP ABAP: Material Master Creation FUNCTION Z_CREATE_MATERIAL_MASTER. DATA: ls_mara TYPE mara, ls_marc TYPE marc, ls_mard TYPE mard. // Set Material Master Data ls_mara-matnr = iv_material_number. ls_mara-mtart = 'FERT'. // Finished Product ls_mara-mbrsh = 'M'. // Industry Sector ls_mara-mtpos_mara = 'NORM'. // Set Plant Data ls_marc-matnr = iv_material_number. ls_marc-werks = iv_plant. ls_marc-mmsta = '01'. // Material Status // Set Storage Location Data ls_mard-matnr = iv_material_number. ls_mard-werks = iv_plant. ls_mard-lgort = iv_storage_location. // Insert Material Master INSERT mara FROM ls_mara. INSERT marc FROM ls_marc. INSERT mard FROM ls_mard. COMMIT WORK. ENDFUNCTION. // SAP ABAP: Purchase Order Creation FUNCTION Z_CREATE_PURCHASE_ORDER. DATA: ls_poheader TYPE bapimepoheader, lt_poitems TYPE TABLE OF bapimepoitem. // Set Purchase Order Header ls_poheader-doc_type = 'NB'. // Standard PO ls_poheader-vendor = iv_vendor_number. ls_poheader-purch_org = iv_purchase_org. ls_poheader-pur_group = iv_purchase_group. // Set Purchase Order Items ls_poitem-material = iv_material_number. ls_poitem-plant = iv_plant. ls_poitem-quantity = iv_quantity. ls_poitem-net_price = iv_price. APPEND ls_poitem TO lt_poitems. // Create Purchase Order CALL FUNCTION 'BAPI_PO_CREATE1' EXPORTING poheader = ls_poheader TABLES poitems = lt_poitems IMPORTING expurchaseorder = ev_po_number. ENDFUNCTION.
